
#include <iostream>
#include <vector>
#include <unordered_map>
#include <algorithm> // Для sort
using namespace std;

int main() {
    setlocale(LC_ALL, "Russian");//для того чтобы я мог вывести по русски
    int n;
    cout << "Введите количество элементов в массиве: ";
    cin >> n;

    vector<int> arr(n);
    cout << "Введите элементы массива: ";
    for (int& num : arr) {
        cin >> num;
    }

    unordered_map<int, int> freq;
    for (int num : arr) {
        freq[num]++;
    }
    cout << "Числа, встречающиеся чётное количество раз: ";
    for (int num : arr) {
        if (freq[num] % 2 == 0) {
            cout << num << " ";
        }
    }

    return 0;
}
