
#include <iostream>
#include <vector>
#include <unordered_map>
#include <algorithm> // Для sort
using namespace std;

int main() {
    setlocale(LC_ALL, "Russian");//для того чтобы я мог вывести по русски
    int n;
    cout << "Введите количество элементов в массиве: ";
    cin >> n;

    vector<int> arr(n);           // Создаём вектор размером n
    cout << "Введите элементы массива: "; // для ввода элементов массива
    for (int& num : arr) {        // Проходим по каждому элементу массива
        cin >> num;           
    }

    unordered_map<int, int> freq; // Создаём хэш-таблицу для подсчёта частоты появления каждого числа
    for (int num : arr) {         // Проходим по каждому элементу массива
        freq[num]++;              // Увеличиваем счётчик для текущего числа в хэш-таблице
    }

    cout << "Числа, встречающиеся чётное количество раз: ";
    for (int num : arr) {        // Проходим по каждому элементу массива
        if (freq[num] % 2 == 0) { // Проверяем, если частота текущего числа чётная
            cout << num << " ";   // Выводим число, если его частота чётная
        }
    }

    return 0;                     // Завершаем выполнение программы
}
